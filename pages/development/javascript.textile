h3(#5). Ajax

* *(#5_1)创建XMLHttpRequest对象*

bc. function getXmlHttpRequest(){
    var xmlHttp = null;
    try {
        xmlHttp = new XMLHttpRequest();
    } catch (e) {
        try {
            xmlHttp = new ActiveXObject("Msxml2.XMLHTTP");
        } catch (e) {
            xmlHttp = new ActiveXObject("Microsoft.XMLHTTP");
        }
    }
    return xmlHttp;
}
var xhr = getXmlHttpRequest();

* *(#5_2)XMLHttpRequest 属性*

bc. xhr.onreadystatechange  - 状态监听
xhr.readyState          - 状态
xhr.status              - 响应码
xhr.statusText          - 响应文本
xhr.responseText        - 返回文本内容
xhr.responseXML         - 返回XML内容 MIME=text/xml

* *(#5_3)XMLHttpRequest 方法*

bc. xhr.open()  - 与服务器建立连接
xhr.send()  - 向服务器发送数据

* *(#5_4)XMLHttpRequest 请求*

bc. // POST请求
xhr.open("POST", "url", 是否异步：true/false);
xhr.setRequestHeader("Content-Type", "application/x-www-form-urlencoded;charset=utf-8");
send(data);

bc. // GET请求
xhr.open("GET", "url", 是否异步：true/false);
send(null);

* *(#5_5)XMLHttpRequest 响应*

bc. xhr.onreadystatechange = function() {
    if(xhr.readyState == 4 && xhr.status == 200){
        alert(xhr.responseText);
    }
}
